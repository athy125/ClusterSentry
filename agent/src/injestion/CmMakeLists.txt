cmake_minimum_required(VERSION 3.10)
project(ClusterSentryIngestion CXX)
set(CMAKE_CXX_STANDARD 17)
find_package(ZMQ REQUIRED)
find_library(REDIS_PLUS_PLUS redis++ REQUIRED)
find_library(HIREDIS hiredis REQUIRED)
add_executable(ingestion src/ingestion.cpp)
target_link_libraries(ingestion zmq redis++ hiredis)
target_compile_options(ingestion PRIVATE -O3)
